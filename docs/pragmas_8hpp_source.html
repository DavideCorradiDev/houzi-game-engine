10 // The macros in this file can be called to use pragmas in an almost
11 // platform-indipendent way.
12 // Pragmas are still defined according to a specific compiler, but they will
13 // be ignored if a different compiler is used, resulting in portable code.
14 
15 // clang-format off
16 #if defined(HOU_COMPILER_MINGW) || defined(HOU_COMPILER_GCC) || defined(HOU_COMPILER_CLANG)
17  #define HOU_DO_PRAGMA(x) _Pragma(#x)
18 #elif defined(HOU_COMPILER_MSVC)
19  #define HOU_DO_PRAGMA(x) __pragma(x)
20 #endif
21 
22 #if defined(HOU_COMPILER_MINGW) || defined(HOU_COMPILER_GCC) || defined(HOU_COMPILER_CLANG)
23  #define HOU_PRAGMA_GCC_DIAGNOSTIC_PUSH() HOU_DO_PRAGMA(GCC diagnostic push)
24  #define HOU_PRAGMA_GCC_DIAGNOSTIC_POP() HOU_DO_PRAGMA(GCC diagnostic pop)
25  #define HOU_PRAGMA_GCC_DIAGNOSTIC_IGNORED(w) \
26  HOU_DO_PRAGMA(GCC diagnostic ignored #w)
27  #define HOU_PRAGMA_PACK_PUSH(n) HOU_DO_PRAGMA(pack(push, n))
28  #define HOU_PRAGMA_PACK_POP() HOU_DO_PRAGMA(pack(pop))
29 #else
30  #define HOU_PRAGMA_GCC_DIAGNOSTIC_PUSH()
31  #define HOU_PRAGMA_GCC_DIAGNOSTIC_POP()
32  #define HOU_PRAGMA_GCC_DIAGNOSTIC_IGNORED(warning)
33  #define HOU_PRAGMA_PACK_PUSH(n)
34  #define HOU_PRAGMA_PACK_POP()
35 #endif

39  #define HOU_PRAGMA_MSVC_WARNING_PUSH() HOU_DO_PRAGMA(warning(push))
40  #define HOU_PRAGMA_MSVC_WARNING_POP() HOU_DO_PRAGMA(warning(pop))
41  #define HOU_PRAGMA_MSVC_WARNING_DISABLE(w) \
